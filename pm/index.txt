  <h1>Client Report Manager</h1>
  <label>Report Date: <input type="date" id="rdate"></label>
  <button onclick="generate()">Generate PDF</button>
  <button onclick="send()">Send to Client</button>

  <script>
    // ⬇️ REPLACE THESE TWO ⬇️
    const API_URL = 'https://script.google.com/macros/s/AKfycbwZs4YY4mIvCUQNCK2CLMwcEAvvtYtDFPdpWnffAE6lXEE-j9gInlUlNkexLN1yEWuo/exec';
    const API_KEY = 'my‑secret‑key‑1000127316329';
    // ⬆️ REPLACE THESE TWO ⬆️

    async function generate() {
      const date = document.getElementById('rdate').value;
      const res = await fetch(API_URL, { method:'POST', body: JSON.stringify({key:API_KEY, action:'generateClientPDF', date:date}) });
      const data = await res.json();
      if (data.success) alert('PDF generated. Download: ' + data.downloadUrl);
      else alert('Error: ' + data.error);
    }
    async function send() {
      const date = document.getElementById('rdate').value;
      const res = await fetch(API_URL, { method:'POST', body: JSON.stringify({key:API_KEY, action:'sendToClient', date:date}) });
      const data = await res.json();
      if (data.success) alert('Sent to client.');
      else alert('Error: ' + data.error);
    }
  </script>
